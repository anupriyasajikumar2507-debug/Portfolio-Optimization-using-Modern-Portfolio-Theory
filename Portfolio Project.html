import yfinance as yf
import pandas as pd
import numpy as np

# ---------------------------
# Diversified NIFTY 50 Stocks
# ---------------------------

stocks = {
    "HDFC Bank": "HDFCBANK.NS",
    "Infosys": "INFY.NS",
    "HUL": "HINDUNILVR.NS",
    "Maruti": "MARUTI.NS",
    "Bharti Airtel": "BHARTIARTL.NS",
    "Sun Pharma": "SUNPHARMA.NS",
    "Reliance": "RELIANCE.NS",
    "Adani Ports": "ADANIPORTS.NS",
    "Titan": "TITAN.NS",
    "BEL": "BEL.NS"
}

# Benchmark
market = "^NSEI"

# ---------------------------
# Time Period
# ---------------------------

start_date = "2020-01-01"
end_date = "2025-12-31"

# ---------------------------
# Download Adjusted Prices
# ---------------------------

tickers = list(stocks.values()) + [market]

prices = yf.download(
    tickers,
    start=start_date,
    end=end_date,
    auto_adjust=True,
    progress=False
)["Close"]

# Rename columns

rename_dict = {v: k for k, v in stocks.items()}
rename_dict["^NSEI"] = "NIFTY50"

prices.rename(columns=rename_dict, inplace=True)

# ---------------------------
# Daily Returns
# ---------------------------

daily_returns = prices.pct_change().dropna()

# Separate market returns

market_returns = daily_returns["NIFTY50"]

stock_returns = daily_returns.drop(columns=["NIFTY50"])

# ---------------------------
# Annualized Return
# ---------------------------

annual_returns = stock_returns.mean() * 252

# ---------------------------
# Annualized Volatility
# ---------------------------

annual_volatility = stock_returns.std() * np.sqrt(252)

# ---------------------------
# Covariance Matrix
# ---------------------------

cov_matrix = stock_returns.cov() * 252

# ---------------------------
# Correlation Matrix
# ---------------------------

corr_matrix = stock_returns.corr()

# ---------------------------
# Summary Table
# ---------------------------

summary = pd.DataFrame({
    "Annual Return": annual_returns,
    "Annual Volatility": annual_volatility
})

summary["Sharpe (RF=0)"] = (
    summary["Annual Return"] /
    summary["Annual Volatility"]
)

summary = summary.sort_values(
    by="Sharpe (RF=0)",
    ascending=False
)

# ---------------------------
# Export Everything
# ---------------------------

with pd.ExcelWriter(
    "Portfolio_Optimization_Data.xlsx"
) as writer:

    prices.to_excel(
        writer,
        sheet_name="Prices"
    )

    daily_returns.to_excel(
        writer,
        sheet_name="Daily_Returns"
    )

    summary.to_excel(
        writer,
        sheet_name="Summary"
    )

    cov_matrix.to_excel(
        writer,
        sheet_name="Covariance_Matrix"
    )

    corr_matrix.to_excel(
        writer,
        sheet_name="Correlation_Matrix"
    )

print("Excel file created successfully!")

from scipy.optimize import minimize
import numpy as np

n = len(annual_returns)

def portfolio_vol(weights):
    return np.sqrt(
        np.dot(weights.T,
               np.dot(cov_matrix, weights))
    )

constraints = (
    {'type':'eq',
     'fun': lambda w: np.sum(w)-1}
)

bounds = tuple((0,1) for asset in range(n))

initial_weights = np.array([1/n]*n)

gmvp = minimize(
    portfolio_vol,
    initial_weights,
    method='SLSQP',
    bounds=bounds,
    constraints=constraints
)

gmvp_weights = gmvp.x
